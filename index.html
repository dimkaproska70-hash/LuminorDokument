Обновление 1.1

Добро пожаловать в LuminorLib

Интерфейс получил глобальное обновление. Теперь библиотека поддерживает кастомную графику, интеграцию Platoboost (Key System), анимированные видео-фоны (.mp4) и умную систему динамических иконок, которые автоматически загружаются из вашего репозитория.

Системные требования эксплоита

Для корректной отрисовки фонов и иконок ваш executor обязан поддерживать файловые функции writefile, makefolder, isfile, а также getcustomasset.

Начать: Создание Окна

1. Создание окна, фоны и Platoboost

В метод CreateWindow добавлены новые мощные параметры. Вы можете указать название фона (или написать "none" для отключения фона), а также включить встроенную ключ-систему (true/false) с вашим ID проекта из Platoboost.

Доступные заготовки фонов:

none

Без фона

Pan1.mp4

Анимация

Pan2.png

Статика

Pan3.png

Статика

Pan4.png

Статика

Pan5.png

Статика
LUA 
local Window = LuminorLib:CreateWindow(
    "Luminor Premium", -- Заголовок (Title)
    "Luminor_UI",       -- Имя ScreenGui (CoreGui)
    "by DADILK",        -- Ватермарка
    "Classic",          -- Цветовая тема
    "none",             -- Фон ("Pan1.mp4", "Pan2.png" или "none")
    true,               -- Ключ-система (true = включить, false = выключить)
    "12345"             -- ID проекта на сайте Platoboost
)
Введение Шаг 2: Вкладки

2. Вкладки (Tabs)

Вкладки теперь принимают параметр iconName. Иконка автоматически центрируется вместе с текстом. Если оставить текст пустым (""), то на вкладке будет отображаться только красивая иконка по центру.

LUA 
-- Создаем обычную вкладку с иконкой и текстом
local MainTab = Window:CreateTab("Главная", "home.ico")

-- Создаем компактную вкладку только с иконкой шестеренки
local SettingsTab = Window:CreateTab("", "settings.ico")
Шаг 1: Окно Шаг 3: Переключатели

3. Переключатели (Toggles)

Тогглы теперь обладают умными иконками, которые загораются неоновым цветом при активации функции. Поддерживают пустой текст для центровки иконок. (Примечание: кнопки и тогглы всегда остаются непрозрачными для лучшей видимости).

LUA 
-- Синтаксис: CreateToggle(text, neonColor, defaultState, iconName, callback)
MainTab:CreateToggle("Авто-Фарм", "green", false, "farm.ico", function(state)
    print("Состояние Авто-фарма: " .. tostring(state))
end)

-- Тоггл-кнопка (без текста, только светящийся глаз)
MainTab:CreateToggle("", "purple", false, "eye.ico", function(state)
    print("ESP включен: " .. tostring(state))
end)
Шаг 2: Вкладки Шаг 4: Кнопки

4. Интерактивные Кнопки

Кнопки получили красивый эффект волны (ripple-effect) при нажатии, а также возможность добавлять иконку рядом с текстом или использовать только иконку для минималистичных меню.

LUA 
-- Синтаксис: CreateButton(text, iconName, callback)
MainTab:CreateButton("Телепорт к боссу", "teleport.ico", function()
    print("Инициализация телепортации...")
end)

-- Минималистичная кнопка-иконка (например, для сброса)
MainTab:CreateButton("", "trash.ico", function()
    print("Конфиг очищен!")
end)

5. Готовый шаблон скрипта

Это мощный, готовый к запуску фундамент для вашего скрипта. В нём уже реализованы вкладки, тогглы для Аимбота и Авто-Фарма, телепорты, а также система сохранения конфигурации через файловую систему.

ПОЛНЫЙ СКРИПТ LUA 
local LuminorLib = loadstring(game:HttpGet("https://raw.githubusercontent.com/dimkaproska70-hash/LuminorGui/main/Lib.lua"))()

-- 1. Инициализируем главное окно с премиум темой, отключенным фоном и ключ-системой
local Window = LuminorLib:CreateWindow(
    "Luminor Premium Edition", 
    "LuminorPrivateUI", 
    "by DADILK & Co", 
    "classic", 
    "none",     -- Фон отключен ("none" / "Pan1.mp4")
    true,       -- Ключ-система включена
    "12345"     -- Ваш ID проекта Platoboost
)

-- 2. Вкладки под разные задачи
local MainTab = Window:CreateTab("Главная", "home.ico")
local CombatTab = Window:CreateTab("Аимбот & Бой", "sword.ico")
local VisualsTab = Window:CreateTab("ВХ / Визуалы", "eye.ico")
local TeleportTab = Window:CreateTab("Телепортация", "map.ico")
local ConfigTab = Window:CreateTab("Конфигурация", "settings.ico")

-- Локальные настройки скрипта для отслеживания состояния
local Config = {
    Aimbot = false,
    SilentAim = false,
    EspPlayers = false,
    EspBoxes = false,
    AutoFarm = false,
    SelectedMob = "Mage Level 50"
}

-- ================= [ ГЛАВНАЯ ВКЛАДКА ] =================
MainTab:CreateToggle("Автоматический фарм (Auto-Farm)", "green", false, "farm.ico", function(state)
    Config.AutoFarm = state
    if state then
        task.spawn(function()
            while Config.AutoFarm do
                task.wait(1)
                print("Фармим мобов... Текущая цель: " .. Config.SelectedMob)
            end
        end)
    end
end)

MainTab:CreateButton("Быстрая очистка карты", "trash.ico", function()
    print("Карта очищена от мусорных объектов и эффектов!")
end)

-- ================= [ ВКЛАДКА БОЯ ] =================
CombatTab:CreateToggle("Включить Аимбот", "purple", false, "aim.ico", function(state)
    Config.Aimbot = state
    print("Аимбот изменен на:", state)
end)

CombatTab:CreateToggle("Сайлент Аим (Silent Aim)", "red", false, "crosshair.ico", function(state)
    Config.SilentAim = state
    print("Silent Aim:", state)
end)

-- ================= [ ВКЛАДКА ВИЗУАЛОВ ] =================
VisualsTab:CreateToggle("Подсветка игроков (ESP)", "cyan", false, "user.ico", function(state)
    Config.EspPlayers = state
    print("ESP Игроков:", state)
end)

VisualsTab:CreateToggle("Отображать боксы", "blue", false, "box.ico", function(state)
    Config.EspBoxes = state
    print("Боксы игроков:", state)
end)

-- ================= [ ВКЛАДКА ТЕЛЕПОРТОВ ] =================
TeleportTab:CreateButton("Телепорт в Безопасную зону", "shield.ico", function()
    local lplayer = game.Players.LocalPlayer
    if lplayer and lplayer.Character and lplayer.Character:FindFirstChild("HumanoidRootPart") then
        lplayer.Character.HumanoidRootPart.CFrame = CFrame.new(0, 50, 0)
        print("Успешный телепорт в безопасную зону.")
    end
end)

TeleportTab:CreateButton("Телепорт к Боссу", "teleport.ico", function()
    print("Поиск босса на карте...")
end)

-- ================= [ ВКЛАДКА НАСТРОЕК ] =================
ConfigTab:CreateButton("Сохранить текущие настройки", "save.ico", function()
    if writefile then
        local json = game:GetService("HttpService"):JSONEncode(Config)
        writefile("Luminor_Config.json", json)
        print("Конфиг успешно сохранен в файл 'Luminor_Config.json'!")
    else
        print("Ошибка: Ваш эксплоит не поддерживает writefile!")
    end
end)

ConfigTab:CreateButton("Загрузить настройки", "download.ico", function()
    if isfile and isfile("Luminor_Config.json") then
        local json = readfile("Luminor_Config.json")
        Config = game:GetService("HttpService"):JSONDecode(json)
        print("Конфигурация успешно загружена!")
    else
        print("Ошибка: Файл настроек 'Luminor_Config.json' не найден!")
    end
end)

ConfigTab:CreateButton("Уничтожить интерфейс (Unload)", "power.ico", function()
    local gui = game:GetService("CoreGui"):FindFirstChild("LuminorPrivateUI") or game:GetService("Players").LocalPlayer.PlayerGui:FindFirstChild("LuminorPrivateUI")
    if gui then
        gui:Destroy()
        print("Интерфейс успешно выгружен из памяти игры.")
    end
end)
