Начать: Создание Окна

1. Создание окна, фоны и Platoboost

Метод CreateWindow теперь принимает таблицу опций шестым параметром. Вы можете указать название фона простым словом (например "Pan1" — библиотека сама найдёт подходящий файл .png/.mp4), а также передать platoboostId и platoboostApiKey для автоматического заполнения полей ключ-системы.

Доступные заготовки фонов:

none

Без фона

Pan1

Анимация

Pan2

Статика

Pan3

Статика

Pan4

Статика

Pan5

Статика
LUA 
local Window = LuminorLib:CreateWindow(
    "Luminor Premium",   -- Заголовок (Title)
    "Luminor_UI",        -- Имя ScreenGui (CoreGui)
    "by DADILK",         -- Ватермарка
    "Classic",           -- Цветовая тема
    "Pan1",              -- Фон (можно "none", "Pan2", "Pan3"...)
    {
        keySystem = true,               -- Включить ключ-систему
        platoboostId = "123456",        -- Ваш ID проекта
        platoboostApiKey = "key_abcde"  -- Ваш API ключ
    }
)

Если не указывать platoboostId и platoboostApiKey, поля в окне ключ-системы будут пустыми и их можно заполнить вручную.

Введение Шаг 2: Вкладки

2. Вкладки (Tabs)

Иконки вкладок теперь указываются именем без расширения (например, "home"). Библиотека автоматически найдет файл home.png или home.PNG в репозитории. Можно оставить текст пустым — тогда вкладка будет содержать только иконку по центру.

LUA 
-- Обычная вкладка с иконкой и текстом
local MainTab = Window:CreateTab("Главная", "home")

-- Компактная вкладка только с иконкой шестеренки
local SettingsTab = Window:CreateTab("", "settings")
Шаг 1: Окно Шаг 3: Переключатели

3. Переключатели (Toggles)

Тогглы поддерживают умные иконки — просто передайте имя без расширения, и оно загорится неоновым цветом при активации. Можно оставить текст пустым, чтобы получить компактную кнопку-иконку.

LUA 
-- Синтаксис: CreateToggle(text, neonColor, defaultState, iconName, callback)
MainTab:CreateToggle("Авто-Фарм", "green", false, "farm", function(state)
    print("Состояние Авто-фарма: " .. tostring(state))
end)

-- Тоггл-кнопка без текста (только светящаяся иконка)
MainTab:CreateToggle("", "purple", false, "eye", function(state)
    print("ESP включен: " .. tostring(state))
end)
Шаг 2: Вкладки Шаг 4: Кнопки

4. Интерактивные Кнопки

Кнопки с эффектом волны при нажатии. Иконка задаётся так же — именем без расширения. Если текст не нужен, кнопка отобразится с одной иконкой по центру.

LUA 
-- Синтаксис: CreateButton(text, iconName, callback)
MainTab:CreateButton("Телепорт к боссу", "teleport", function()
    print("Инициализация телепортации...")
end)

-- Минималистичная кнопка-иконка (сброс)
MainTab:CreateButton("", "trash", function()
    print("Конфиг очищен!")
end)

5. Готовый шаблон скрипта

Полный фундамент для вашего скрипта с примерами всех элементов, сохранением конфигурации и новой ключ-системой Platoboost.

ПОЛНЫЙ СКРИПТ LUA 
local LuminorLib = loadstring(game:HttpGet("https://raw.githubusercontent.com/dimkaproska70-hash/LuminorGui/main/Lib.lua"))()

-- 1. Инициализация окна с ключ-системой и фоном Pan1
local Window = LuminorLib:CreateWindow(
    "Luminor Premium Edition",
    "LuminorPrivateUI",
    "by DADILK & Co",
    "classic",
    "Pan1",   -- Фон (автоматически найдёт Pan1.mp4 / Pan1.png)
    {
        keySystem = true,
        platoboostId = "123456",          -- Ваш ID проекта
        platoboostApiKey = "your_api_key" -- Ваш API ключ
    }
)

-- 2. Вкладки (иконки без расширений)
local MainTab = Window:CreateTab("Главная", "home")
local CombatTab = Window:CreateTab("Аимбот & Бой", "sword")
local VisualsTab = Window:CreateTab("ВХ / Визуалы", "eye")
local TeleportTab = Window:CreateTab("Телепортация", "map")
local ConfigTab = Window:CreateTab("Конфигурация", "settings")

local Config = {
    Aimbot = false,
    SilentAim = false,
    EspPlayers = false,
    EspBoxes = false,
    AutoFarm = false,
    SelectedMob = "Mage Level 50"
}

-- ================= [ ГЛАВНАЯ ВКЛАДКА ] =================
MainTab:CreateToggle("Авто-Фарм", "green", false, "farm", function(state)
    Config.AutoFarm = state
    if state then
        task.spawn(function()
            while Config.AutoFarm do
                task.wait(1)
                print("Фармим... Текущая цель: " .. Config.SelectedMob)
            end
        end)
    end
end)

MainTab:CreateButton("Очистить карту", "trash", function()
    print("Карта очищена!")
end)

-- ================= [ ВКЛАДКА БОЯ ] =================
CombatTab:CreateToggle("Аимбот", "purple", false, "aim", function(state)
    Config.Aimbot = state
    print("Аимбот:", state)
end)

CombatTab:CreateToggle("Сайлент Аим", "red", false, "crosshair", function(state)
    Config.SilentAim = state
    print("Silent Aim:", state)
end)

-- ================= [ ВКЛАДКА ВИЗУАЛОВ ] =================
VisualsTab:CreateToggle("ESP игроков", "cyan", false, "user", function(state)
    Config.EspPlayers = state
    print("ESP Игроков:", state)
end)

VisualsTab:CreateToggle("Боксы", "blue", false, "box", function(state)
    Config.EspBoxes = state
    print("Боксы:", state)
end)

-- ================= [ ТЕЛЕПОРТЫ ] =================
TeleportTab:CreateButton("В безопасную зону", "shield", function()
    local lp = game.Players.LocalPlayer
    if lp and lp.Character and lp.Character:FindFirstChild("HumanoidRootPart") then
        lp.Character.HumanoidRootPart.CFrame = CFrame.new(0, 50, 0)
        print("Телепорт выполнен!")
    end
end)

TeleportTab:CreateButton("К боссу", "teleport", function()
    print("Поиск босса...")
end)

-- ================= [ КОНФИГУРАЦИЯ ] =================
ConfigTab:CreateButton("Сохранить", "save", function()
    if writefile then
        writefile("Luminor_Config.json", game:GetService("HttpService"):JSONEncode(Config))
        print("Конфиг сохранён!")
    else
        print("Ошибка: writefile недоступен")
    end
end)

ConfigTab:CreateButton("Загрузить", "download", function()
    if isfile and isfile("Luminor_Config.json") then
        Config = game:GetService("HttpService"):JSONDecode(readfile("Luminor_Config.json"))
        print("Конфиг загружен!")
    else
        print("Файл конфига не найден")
    end
end)

ConfigTab:CreateButton("Выход", "power", function()
    local gui = game:GetService("CoreGui"):FindFirstChild("LuminorPrivateUI")
    if gui then gui:Destroy() print("Интерфейс выгружен") end
end)
